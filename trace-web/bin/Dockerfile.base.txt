FROM ubuntu:20.04

ENV DEBIAN_FRONTEND=noninteractive
RUN apt-get update \
    && apt-get install -y --no-install-recommends --fix-missing tzdata language-pack-zh-hans logrotate curl gettext-base gnupg2 lsb-core certbot \
    && ln -snf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime && echo Asia/Shanghai > /etc/timezone 

RUN curl -k https://openresty.org/package/pubkey.gpg | apt-key add - \
    && echo "deb http://openresty.org/package/ubuntu $(lsb_release -cs) main" | tee /etc/apt/sources.list.d/openresty.list \
    && apt-get update \
    && apt-get install -y --no-install-recommends --fix-missing openresty
